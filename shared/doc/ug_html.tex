%% User-callable function
%%---------------------------------
\newcommand{\ucfunction}[6]{
  \begin{itemize}
  \item[\textcolor{darkgreen}{\bf #1}]{}
  \item[Call]{\id{#2}}
  \item[Description]{#3}
  \item[Arguments]{#4}
  \item[Return value]{#5}
  \item[Notes]{#6}
  \end{itemize}
}
%%---------------------------------
%% User-supplied function
%%---------------------------------
\newcommand{\usfunction}[6]{
  \begin{itemize}
  \item[\textcolor{darkgreen}{\bf #1}]{}
  \item[Definition]{\id{\begin{tabular}[t]{@{}r@{}l@{}}#2\end{tabular}}}
  \item[Purpose]{#3}
  \item[Arguments]{#4}
  \item[Return value]{#5}
  \item[Notes]{#6}
  \end{itemize}
}
